def svg_url_for_otp_credential(otp_credential)
  qrcode = qrcode(otp_credential)
  qrcode.as_svg({
    offset: 0,
    color: '000',
    shape_rendering: 'crispEdges',
    module_size: 3,
    standalone: true      
  }).html_safe
end
