def copy_template_file
  @namespace = class_name.split("::")
  @namespace.pop
  @namespace = @namespace.join("::")
  template "coffeescript.coffee.erb", file_path
end

def namespace_path
  if namespace.blank?
    return ""
  else
    namespace.split("::").map(&:underscore).join("/") + "/"
  end
end
